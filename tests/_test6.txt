_simpleInterest
type real: principal;
type real: rate;
type real: time;
type real: interest;
principal <--- 5000.0;
rate <--- 0.05;
time <--- 2.5;
interest <--- (principal * rate * time);
return [interest];
end

_main
[totalInterest] <--- call _simpleInterest with parameters [];
write(totalInterest);
