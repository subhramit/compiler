% PrimeCheck.txt
type int: num
type int: isPrime;
type int: i;
num <--- 17;  % Change this value for different test cases
isPrime <--- 1;

i <--- 2;
while (i < num)
    if (num % i == 0) then
        isPrime <--- 0;
        break;
    endif
    i <--- i + 1;
endwhile

if (isPrime) then
    write(num);
else
    write(num-1);
endif
