record #outer 
    type int: outerVar;
    record #inner
        type real: innerVar;
    endrecord
endrecord

int: result;
[result] <--- call calculate with parameters [10, 3.14];
write(result);

int: calculate(int x, real y)
    record #outer obj;
    obj.outerVar <--- x;
    obj.inner.innerVar <--- y;
    return [obj.outerVar + obj.inner.innerVar];
end
